Block Scope
Block-Scoped Variables
Block-scoped variables without variable hoisting.
ECMAScript 6 — style: modernized | traditional
for (let i = 0; i < a.length; i++) { let x = a[i]; … } for (let i = 0; i < b.length; i++) let y = b[i]; … }
ECMAScript 5 — style: modernized | traditional
var i, x, y; for (i = 0; i < a.length; i++) { x = a[i]; … } for (i = 0; i < b.length; i++) y = b[i]; … }
Block Scope
Block-Scoped Constants
Block-scoped constants without constant hoisting.
ECMAScript 6 — style: modernized | traditional
const pi = 3.141593; pi > 3.0;
ECMAScript 5 — style: modernized | traditional
var pi = (function () { var pi = 3.141593; return function () { return pi }; })(); pi() > 3.0;
Arrow Functions
Expression Bodies
More expressive closure syntax.
ECMAScript 6 — style: modernized | traditional
odds = evens.map(v => v + 1); pairs = evens.map(v => ({ even: v, odd: v + 1 })); nums = evens.map((v, i) => v + i);
ECMAScript 5 — style: modernized | traditional
odds = evens.map(function (v) { return v + 1; }); pairs = evens.map(function (v) { return { even: v, odd: v + 1 }; }); nums = evens.map(function (v, i) { return v + i; });
Arrow Functions
Statement Bodies
More expressive closure syntax.
ECMAScript 6 — style: modernized | traditional
nums.forEach(v => { if (v % 5 === 0) fives.push(v); })
ECMAScript 5 — style: modernized | traditional
nums.forEach(function (v) { if (v % 5 === 0) fives.push(v); });
Arrow Functions
Lexical this
More intuitive handling of current object context.
ECMAScript 6 — style: modernized | traditional
this.nums.forEach((v) => { if (v % 5 === 0) this.fives.push(v); });
ECMAScript 5 — style: modernized | traditional
var self = this this.nums.forEach(function (v) { if (v % 5 === 0) self.fives.push(v); });
Classes
Class Definition
More intuitive, OOP-style and boilerplate-free classes.
ECMAScript 6 — style: modernized | traditional
class Shape { constructor (id, x, y) { this.id = id; this.move(x, y); } move (x, y) { this.x = x; this.y = y; } }
ECMAScript 5 — style: modernized | traditional
var Shape = function (id, x, y) { this.id = id; this.move(x, y); }; Shape.prototype.move = function (x, y) { this.x = x; this.y = y; };
Classes
Class Inheritance
More intuitive, OOP-style and boilerplate-free inheritance.
ECMAScript 6 — style: modernized | traditional
class Rectangle extends Shape { constructor (id, x, y, width, height) { super(id, x, y); this.width = width; this.height = height; } } class Circle extends Shape { constructor (id, x, y, radius) { super(id, x, y); this.radius = radius; } }
ECMAScript 5 — style: modernized | traditional
var Rectangle = function (id, x, y, width, height) { Shape.call(this, id, x, y); this.width = width; this.height = height; }; Rectangle.prototype = Object.create(Shape.prototype); Rectangle.prototype.constructor = Rectangle; var Circle = function (id, x, y, radius) { Shape.call(this, id, x, y); this.radius = radius; }; Circle.prototype = Object.create(Shape.prototype); Circle.prototype.constructor = Circle;
Classes
Static Members
Simple support for static class members.
ECMAScript 6 — style: modernized | traditional
class Rectangle extends Shape { … static defaultRectangle () { return new Rectangle("default", 0, 0, 100, 100); } } class Circle extends Shape { … static defaultCircle () { return new Circle("default", 0, 0, 100); } }
ECMAScript 5 — style: modernized | traditional
var Rectangle = function (id, x, y, width, height) { … }; Rectangle.defaultRectangle = function () { return new Rectangle("default", 0, 0, 100, 100); }; var Circle = function (id, x, y, width, height) { … }; Circle.defaultCircle = function () { return new Circle("default", 0, 0, 100); };
Classes
Getter/Setter
More intuitive and boilerplate-free getter/setter members.
ECMAScript 6 — style: modernized | traditional
class Shape { … get area { throw new Error("calculated area not implemented"); } set area { throw new Error("cannot set calculated area"); } } class Rectangle extends Shape { … get area { return this.width * this.height; } } class Circle extends Shape { … get area { return Math.PI * this.radius * this.radius; } } let rec = new Rectangle("foo", 0, 0, 50, 20); circle.area === 1000; circle.area = 2000; // exception!
ECMAScript 5 — style: modernized | traditional
Shape.prototype.setArea = function (area) { throw new Error("cannot set calculated area"); }; Shape.prototype.getArea = function () { throw new Error("calculated area not implemented"); }; Rectangle.prototype.getArea = function () { return this.width * this.height; }; Circle.prototype.getArea = function () { return Math.PI * this.radius * this.radius; }; var rec = new Rectangle("foo", 0, 0, 50, 20); circle.getArea() === 1000; circle.setArea(2000); // exception!
Enhanced Object Literals
Property Shorthand
Shorter syntax for common object property definition idiom.
ECMAScript 6 — style: modernized | traditional
obj = { x, y };
ECMAScript 5 — style: modernized | traditional
obj = { x: x, y: y };
Enhanced Object Literals
Computed Property Names
Support for computed names in object property definitions.
ECMAScript 6 — style: modernized | traditional
obj = { foo: "bar", [ "prop_" + foo() ]: 42 };
ECMAScript 5 — style: modernized | traditional
obj = { foo: "bar" }; obj[ "prop_" + foo() ] = 42;
Modules
Symbol Export/Import
Support for exporting/importing symbols from/to modules without global namespace pollution.
ECMAScript 6 — style: modernized | traditional
// lib/math.js export function sum (x, y) { return x + y }; export var pi = 3.141593; // someApp.js import * as math from "lib/math"; alert("2 = " + math.sum(math.pi, math.pi)); // otherApp.js import { sum, pi } from "lib/math";; alert("2 = " + sum(pi, pi));
ECMAScript 5 — style: modernized | traditional
// lib/math.js LibMath = {}; LibMath.sum = function (x, y) { return x + y }; LibMath.pi = 3.141593; // someApp.js var math = LibMath; alert("2 = " + math.sum(math.pi, math.pi)); // otherApp.js var sum = LibMath.sum, pi = LibMath.pi; alert("2 = " + sum(pi, pi));
Modules
Default & Wildcard
Marking a symbol as the default exported symbol and mass-mixin of symbols.
ECMAScript 6 — style: modernized | traditional
// lib/mathplusplus.js export * from "lib/math"; export var e = 2.71828182846; export default function (x) { return Math.exp(x) }; // someApp.js import exp, { pi, e } from "lib/mathplusplus"; alert("2 = " + exp(pi, e));
ECMAScript 5 — style: modernized | traditional
// lib/mathplusplus.js LibMathPP = {}; for (symbol in LibMath) if (LibMath.hasOwnProperty(symbol)) LibMathPP[symbol] = LibMath[symbol]; LibMathPP.e = 2.71828182846; LibMathPP.exp = function (x) { return Math.exp(x) }; // someApp.js var exp = LibMathPP.exp, pi = LibMathPP.pi, e = libMathPP.e; alert("2 = " + exp(pi, e));
Template Strings
String Interpolation
Intuitive expression interpolation for strings.
ECMAScript 6 — style: modernized | traditional
mail = `Hello ${name}, let's eat ${num +1} pizzas.`;
ECMAScript 5 — style: modernized | traditional
mail = "Hello " + name + ", let's eat " + (num + 1) + "pizzas.";
Template Strings
Custom Interpolation
Flexible expression interpolation for arbitrary methods.
ECMAScript 6 — style: modernized | traditional
get`http://example.com/foo?bar=${bar}&quux=${quux}`;
ECMAScript 5 — style: modernized | traditional
get([ "http://example.com/foo?bar=", "&quux=" ], [ bar, quux ]);
Destructuring
Array Matching
Intuitive and flexible destructuring of Arrays into individual variables during assignment.
ECMAScript 6 — style: modernized | traditional
var list = [ 1, 2, 3 ]; var [ a, , b ] = list;
ECMAScript 5 — style: modernized | traditional
var list = [ 1, 2, 3 ]; var a = list[0]; var b = list[2];
Destructuring
Object Matching with Shorthand Notation
Intuitive and flexible destructuring of Objects into individual variables during assignment.
ECMAScript 6 — style: modernized | traditional
var { op, lhs, rhs } = getASTNode();
ECMAScript 5 — style: modernized | traditional
var tmp = getASTNode(); var op = tmp.op; var lhs = tmp.lhs; var rhs = tmp.rhs;
Destructuring
Object Matching with Deeper Matching
Intuitive and flexible destructuring of Objects into individual variables during assignment.
ECMAScript 6 — style: modernized | traditional
var { op: a, lhs: { op: b }, rhs: c } = getASTNode();
ECMAScript 5 — style: modernized | traditional
var tmp = getASTNode(); var a = tmp.op; var b = tmp.lhs.op; var c = tmp.rhs;
Destructuring
Object Matching in Parameter Position
Intuitive and flexible destructuring of Objects into individual parameters during function calls.
ECMAScript 6 — style: modernized | traditional
function g ({ name: x }) { console.log(x); } g({ name: 5 });
ECMAScript 5 — style: modernized | traditional
function g (arg) { var x = arg.name; console.log(x); }; g({ name: 5 });
Extended Parameter Handling
Default Parameter Values
Simple and intuitive default values for function parameters.
ECMAScript 6 — style: modernized | traditional
function f (x, y = 7, z = 42) { return x + y + z; } f(1) === 50;
ECMAScript 5 — style: modernized | traditional
function f (x, y, z) { if (typeof y === "undefined") y = 7; if (typeof z === "undefined") z = 42; return x + y + z; }; f(1) === 50;
Extended Parameter Handling
Rest Parameter
Aggregation of remaining arguments into single parameter of variadic functions.
ECMAScript 6 — style: modernized | traditional
function f (x, y, …a) { return (x + y) * a.length; } f(1, 2, "hello", true, 7) === 9;
ECMAScript 5 — style: modernized | traditional
function f (x, y) { var a = Array.prototype.slice.call(arguments, 2); return (x + y) * a.length; }; f(1, 2, "hello", true, 7) === 9;
Extended Parameter Handling
Spread Operator
Spreading of array elements into individual parameters of a function.
ECMAScript 6 — style: modernized | traditional
var params = [ "hello", true, 7 ]; f(1, 2, …params) === 9;
ECMAScript 5 — style: modernized | traditional
var params = [ "hello", true, 7 ]; f.apply(undefined, [ 1, 2 ].concat(params)) === 9;
Iterators
Iterator & For-Of Operator
Support "iterable" protocol to allow objects to customize their iteration behaviour. Additionally, support "iterator" protocol to produce sequence of values (either finite or infinite). Finally, provide convenient of operator to iterate over all values of an iterable object.
ECMAScript 6 — style: modernized | traditional
let fibonacci = { [Symbol.iterator]() { let pre = 0, cur = 1; return { next () { [ pre, cur ] = [ cur, pre + cur ]; return { done: false, value: cur }; } }; } } for (let n of fibonacci) { if (n > 1000) break; console.log(n); }
ECMAScript 5 — style: modernized | traditional
var fibonacci = { next: ((function () { var pre = 0, cur = 1; return function () { tmp = pre; pre = cur; cur += tmp; return cur; }; })(); }; var n; for (;;) { n = fibonacci.next(); if (n > 1000) break; console.log(n); }
Iterators
Generators
Support for generators, a special case of Iterators where the control flow can be paused and resumed. Beside iteration, supports asynchronous programming.
ECMAScript 6 — style: modernized | traditional
let fibonacci = { [Symbol.iterator]: function* () { let pre = 0, cur = 1; for (;;) { [ pre, cur ] = [ cur, pre + cur ]; yield cur; } } } for (let n of fibonacci) { if (n > 1000) break; console.log(n); }
ECMAScript 5 — style: modernized | traditional
var fibonacci = { next: ((function () { var pre = 0, cur = 1; return function () { tmp = pre; pre = cur; cur += tmp; return cur; }; })(); }; var n; for (;;) { n = fibonacci.next(); if (n > 1000) break; console.log(n); }

Map Data-Structure
Cleaner data-structure for common algorithms based on maps.
ECMAScript 6 — style: modernized | traditional
let m = new Map(); m.set("hello", 42); m.set(s, 34); m.get(s) === 34; m.size === 2; for (let [ key, val ] of m.entries()) console.log(key + " = " + val);
